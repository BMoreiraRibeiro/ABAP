    price_unit = '1'
  ) TO lt_items.

  APPEND VALUE #(
    po_item   = '00020'
    material  = 'X'
    plant     = 'X'
    quantity  = 'X'
    net_price = 'X'
    price_unit = 'X'
  ) TO lt_itemsx.

  " Criar pedido
  CALL FUNCTION 'BAPI_PO_CREATE1'
    EXPORTING
      poheader  = ls_header
      poheaderx = ls_headerx
    IMPORTING
      exppurchaseorder = lv_ponumber
    TABLES
      return    = lt_return
      poitem    = lt_items
      poitemx   = lt_itemsx.

  " Verificar resultado
  READ TABLE lt_return WITH KEY type = 'E' TRANSPORTING NO FIELDS.

  IF sy-subrc = 0.
    " Erro
    WRITE: / '‚ùå Erro ao criar pedido:', /.
    LOOP AT lt_return INTO DATA(ls_msg) WHERE type CA 'EA'.
      WRITE: / ls_msg-message.
    ENDLOOP.

    CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.

  ELSE.
    " Sucesso
    CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'
      EXPORTING wait = 'X'.

    WRITE: / '‚úÖ Pedido criado com sucesso!'.
    WRITE: / |N√∫mero: { lv_ponumber }|.
    WRITE: / |Itens: { lines( lt_items ) }|, /.

    " Mensagens de sucesso
    LOOP AT lt_return INTO ls_msg WHERE type = 'S'.
      WRITE: / ls_msg-message.
    ENDLOOP.
  ENDIF.
